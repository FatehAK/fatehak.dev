const getHashData = (array, key) => {
  const hashVal = (str) => {
      let hash = 0;
      for (let i = 0; i < str.length; i++) {
          let charCode = str.charCodeAt(i);
          hash += charCode;
      }
      return hash;
  };
  const hash = hashVal(key);
  const index = hash % array.length;
  return array[index];
};
